'pinned dark:pinned' : 'hover:bg-gray-100 dark:hover:bg-gray-700'}`; const avatarSrc = contact.avatarUrl ? contact.avatarUrl : `https://ui-avatars.com/api/?name=${contact.nickname.charAt(0)}&background=random&size=64`; contactEl.innerHTML = `
${contact.nickname}
${contact.status || 'Offline'}
`; const pinBtn = contactEl.querySelector('.pin-btn'); if (pinBtn) { pinBtn.addEventListener('click', togglePin); // Ab togglePin defined hai } return contactEl; } function renderContact(contact) { const contactEl = createContactCard(contact); // Ab createContactCard defined hai contactsListDiv.appendChild(contactEl); } // Bell icon par count update karna function updateRequestCount(count) { if (count > 0) { requestsCount.textContent = count; requestsCount.classList.remove('hidden'); } else { requestsCount.classList.add('hidden'); } } // Accept/Decline button ki request bhejna async function handleRequestAction(event) { const button = event.target; // Sirf 'accept-btn' ya 'decline-btn' par hi chalao const action = button.classList.contains('accept-btn') ? No contacts found matching "${searchQuery}".

`; } else { contactsListDiv.innerHTML = '

No contacts yet. Tap + to send a request.

'; } return; } contacts.forEach(renderContact); // Ab 'renderContact' defined hai // Pin button listeners ko yahan se hata diya, kyunki 'createContactCard' mein add ho gaya hai } catch (error) { console.error('Error loading contacts:', error); skeletonLoader.style.display = 'none'; contactsListDiv.innerHTML = '

Could not load contacts.

'; } } async function loadFriendRequests() { try { requestsList.innerHTML = '

Loading requests...

${request.displayName}

Could not load requests.

'; } } // 4. Socket Listeners socket.on('new_friend_request', (request) => { console.log('Nayi friend request aayi:', request); showToast(`${request.displayName} sent you a friend request!`); updateRequestCount(requestsCount.textContent + 1); }); socket.on('request_accepted', (newContact) => { console.log('Hamari request accept ho gayi:', newContact); showToast(`${newContact.nickname} accepted your friend request!`); loadContacts(); }); socket.on('friend status', (data) => { const statusDiv = document.getElementById(`status-${data.email}`); if (statusDiv) { const dot = statusDiv.querySelector('span:first-child'); const text = statusDiv.querySelector('span:last-child'); if (data.status === 'Online') { dot.className = 'h-3 w-3 rounded-full bg-green-500'; text.className = 'text-sm text-green-600'; text.textContent = 'Online'; } else { dot.className = 'h-3 w-3 rounded-full bg-gray-400'; text.className = 'text-sm text-gray-500 dark:text-gray-400'; text.textContent = 'Offline'; } } }); // 5.
